; F=5*12+2222/4
; умножения и деления выполнять с помощью операций сдвига. Проверка на четность либо нечетность
; и знак (если числа со знаком) числа осуществлять с помощью логических побитовых команд.

mov ax,12
mov bx,ax
shl ax,1
shl ax,1
add ax,bx ; 5 * 12
mov dx,2222
mov cl,2
shr dx,cl ; 2222/4
add ax,dx ; 5*12 + 2222/4
mov [200],ah
mov [201],al
